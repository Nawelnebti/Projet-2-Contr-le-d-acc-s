#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN 10
#define RST_PIN 9
 
MFRC522 rfid(SS_PIN, RST_PIN);

// Définition des badges autorisés
byte badge1[] = {149, 174, 170, 165};
byte badge2[] = {229, 72, 176, 79};

void setup() {
  Serial.begin(9600);
  SPI.begin();      // Init SPI bus
  rfid.PCD_Init();  // Init MFRC522
  
  Serial.println(F("Système prêt. Scannez un badge..."));
}

void loop() {
  readRFID();
  delay(200);
}

void readRFID() { 
  // Cherche une carte
  if (!rfid.PICC_IsNewCardPresent()) 
    return;

  // Lit le numéro de série
  if (!rfid.PICC_ReadCardSerial())
    return;

  // Vérification de l'autorisation
  if (checkAccess(rfid.uid.uidByte, rfid.uid.size)) {
    Serial.println(F("ACCÈS AUTORISÉ"));
    // Ajoute ici l'action à effectuer (ouvrir un relais, allumer une LED verte, etc.)
  } else {
    Serial.print(F("ACCÈS REFUSÉ ! Badge inconnu : "));
    printDec(rfid.uid.uidByte, rfid.uid.size);
    Serial.println();
  }
   
  // Arrête la lecture
  rfid.PICC_HaltA();
  rfid.PCD_StopCrypto1();
}

/**
 * Compare le badge scanné avec les badges autorisés
 */
bool checkAccess(byte *scannedUID, byte size) {
  bool match1 = true;
  bool match2 = true;

  for (byte i = 0; i < size; i++) {
    if (scannedUID[i] != badge1[i]) match1 = false;
    if (scannedUID[i] != badge2[i]) match2 = false;
  }

  return (match1 || match2);
}

/**
 * Affiche l'UID en décimal dans le moniteur série
 */
void printDec(byte *buffer, byte bufferSize) {
  for (byte i = 0; i < bufferSize; i++) {
    Serial.print(buffer[i] < 0x10 ? " 0" : " ");
    Serial.print(buffer[i], DEC);
  }
}
